--seq
CREATE SEQUENCE New_installments_paid_seq
    INCREMENT BY 1
    START WITH 1
    MINVALUE 1
    MAXVALUE 100000;
    
------
CREATE OR REPLACE TRIGGER INSTALLMENT_ID_TRG
   BEFORE INSERT
   ON HR.INSTALLMENTS_PAID
   REFERENCING NEW AS New OLD AS Old
   FOR EACH ROW
BEGIN
   :NEW.Installment_ID :=New_installments_paid_seq.NEXTVAL;
END INSTALLMENT_ID_TRG;
